# Manual Docker Hub Push Commands
# Run these commands on your dev server

# 1. Login to Docker Hub
echo "YOUR_DOCKER_HUB_ACCESS_TOKEN" | docker login -u runaii98 --password-stdin

# 2. List existing images to see what we have
docker images | grep -E "(runai|postgres|openwebui)"

# 3. Tag and push PostgreSQL image (adjust image name as needed)
docker tag runai-postgres:latest runaii98/runai-postgres:latest
docker push runaii98/runai-postgres:latest

# 4. Tag and push All-in-One image (adjust image name as needed)  
docker tag runai-allinone:latest runaii98/runai-allinone:latest
docker push runaii98/runai-allinone:latest
